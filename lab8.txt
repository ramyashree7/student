create table salesman(salesman_id integer primary key,name varchar(10),city varchar(10),commission integer);
create table customer(customer_id integer primary key,cust_name varchar(10),city varchar(10),grade varchar(1),salesman_id integer,foreign key(salesman_id) references salesman(salesman_id)on delete cascade);
create table orders(ord_no integer primary key,purchase_amt integer,ord_date date,customer_id integer,salesman_id integer,foreign key(salesman_id) references salesman(salesman_id)on delete cascade,foreign key(customer_id) references customer(customer_id)on delete cascade);
